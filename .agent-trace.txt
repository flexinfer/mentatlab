CACHE_REPO="registry.harbor.lan/library/build-cache"

build_image() {
  local name="$1" context="$2"
  echo "=== Building ${name} from ${context} ==="

  local output="type=image,push=true,compression=zstd,compression-level=3"
  output="${output},name=${REGISTRY}/${name}:${CI_COMMIT_SHORT_SHA}"
  if [ "$CI_COMMIT_BRANCH" = "main" ]; then
    output="${output},name=${REGISTRY}/${name}:latest"
  fi

  # First attempt: build with cache
  if buildctl --addr "${BUILDKIT_HOST}" build \
    --frontend dockerfile.v0 \
    --local "context=${context}" \
    --local "dockerfile=${context}" \
    --opt platform=linux/amd64 \
    --opt "build-arg:GOPROXY=${GOPROXY:-}" \
    --opt "build-arg:npm_config_registry=${npm_config_registry:-}" \
    --opt "build-arg:NPM_CONFIG_REGISTRY=${NPM_CONFIG_REGISTRY:-}" \
    --opt "build-arg:PIP_INDEX_URL=${PIP_INDEX_URL:-}" \
    --opt "build-arg:PIP_TRUSTED_HOST=${PIP_TRUSTED_HOST:-}" \
    --export-cache "type=registry,ref=${CACHE_REPO}:${name},mode=max,image-manifest=true" \
    --import-cache "type=registry,ref=${CACHE_REPO}:${name}" \
    --output "${output}"; then
    echo "=== ${name} built successfully ==="
    return 0
  fi

  # Second attempt: retry without import-cache (handles cold/corrupt cache)
  echo "WARNING: ${name} build failed with cache, retrying without cache..."
  buildctl --addr "${BUILDKIT_HOST}" build \
    --frontend dockerfile.v0 \
    --local "context=${context}" \
    --local "dockerfile=${context}" \
    --opt platform=linux/amd64 \
    --opt "build-arg:GOPROXY=${GOPROXY:-}" \
    --opt "build-arg:npm_config_registry=${npm_config_registry:-}" \
    --opt "build-arg:NPM_CONFIG_REGISTRY=${NPM_CONFIG_REGISTRY:-}" \
    --opt "build-arg:PIP_INDEX_URL=${PIP_INDEX_URL:-}" \
    --opt "build-arg:PIP_TRUSTED_HOST=${PIP_TRUSTED_HOST:-}" \
    --export-cache "type=registry,ref=${CACHE_REPO}:${name},mode=max,image-manifest=true" \
    --output "${output}"
  echo "=== ${name} built successfully (no-cache retry) ==="
}[0;m
